docker build --platform linux/amd64 -t vatsank/hotelservice:2.0 .

Create Container registry with name vats2026
az acr login --name vats2026

docker tag vatsank/hotelservice:2.0 vats2026.azurecr.io/hotelservice:v1

docker push vats2026.azurecr.io/hotelservice:v1

az acr repository list --name vats2026 --output table

az appservice plan create --name hotelservice-plan --resource-group myrg --sku B1 --is-linux --location southindia

ACR_PASSWORD=$(az acr credential show --name vats2026 --query "passwords[0].value" -o tsv)

# Update the Web App with the credentials
az webapp config container set \
  --name vats-hotelservice-app \
  --resource-group myrg \
  --docker-custom-image-name vats2026.azurecr.io/hotelservice:latest \
  --docker-registry-server-url https://vats2026.azurecr.io \
  --docker-registry-server-user vats2026 \
  --docker-registry-server-password $ACR_PASSWORD



az webapp log config --name vats-hotelservice-app --resource-group myrg --docker-container-logging filesystem

az webapp log tail --name vats-hotelservice-app --resource-group myrg

az webapp restart --name vats-hotelservice-app --resource-group myrg
